// Fill out your copyright notice in the Description page of Project Settings.

#pragma once

#include "CoreMinimal.h"
#include "AttributeSet.h"
#include "AbilitySystemComponent.h"
#include "AttributeSet_basic.generated.h"



/**
 * 
 */

UCLASS()
class GAMEPLAY_GAS_API UAttributeSet_basic : public UAttributeSet
{
	GENERATED_BODY()

public:
	//把以下内容定义成为U构造函数方便cpp使用
	UAttributeSet_basic();
	
	
	//健康
	//属性设置为蓝图只读，分配类别为Attributes 设置多人游戏的可复制 ReplicatedUsing
	UPROPERTY(BlueprintReadOnly, Category = "Attributes",ReplicatedUsing=OnRep_Health)
	FGameplayAttributeData Health;
	//设置属性访问器：方便在蓝图中 set get 访问 
	ATTRIBUTE_ACCESSORS_BASIC(UAttributeSet_basic, Health);
	
	//健康最大值【血条上限】
	UPROPERTY(BlueprintReadOnly, Category = "Attributes",ReplicatedUsing=OnRep_MaxHealth)
	FGameplayAttributeData MaxHealth;
	ATTRIBUTE_ACCESSORS_BASIC(UAttributeSet_basic, MaxHealth);
	
	//耐力属性
	UPROPERTY(BlueprintReadOnly, Category = "Attributes",ReplicatedUsing=OnRep_Stamina)
	FGameplayAttributeData Stamina;
	ATTRIBUTE_ACCESSORS_BASIC(UAttributeSet_basic, Stamina);
	
	//耐力最大值【耐力上限】
	UPROPERTY(BlueprintReadOnly, Category = "Attributes",ReplicatedUsing=OnRep_MaxStamina)
	FGameplayAttributeData MaxStamina;	
	ATTRIBUTE_ACCESSORS_BASIC(UAttributeSet_basic, MaxStamina);
	
public:
	//定义一下上面的 OnRep_类别
	UFUNCTION()
	//定义函数输入的是健康状态的老值
	void OnRep_Health(const FGameplayAttributeData& OLDValue) const
	{
		//设置游戏玩法属性通知宏，利用函数处理多角色的复制操作
		GAMEPLAYATTRIBUTE_REPNOTIFY(UAttributeSet_basic, Health, OLDValue);
	}

	UFUNCTION()
	void OnRep_MaxHealth(const FGameplayAttributeData& OLDValue) const
	{
		GAMEPLAYATTRIBUTE_REPNOTIFY(UAttributeSet_basic, MaxHealth, OLDValue);
	}
	
	UFUNCTION()
	void OnRep_Stamina(const FGameplayAttributeData& OLDValue) const
	{
		GAMEPLAYATTRIBUTE_REPNOTIFY(UAttributeSet_basic, Stamina, OLDValue);
	}
	
	UFUNCTION()
	void OnRep_MaxStamina(const FGameplayAttributeData& OLDValue) const
	{
		GAMEPLAYATTRIBUTE_REPNOTIFY(UAttributeSet_basic, MaxStamina, OLDValue);
	}
	
	//获取生命周期复现的状态：决定我们该任何复制操作，如何复制生命体状态。
	virtual void GetLifetimeReplicatedProps(TArray<class FLifetimeProperty>& OutLifetimeProps) const override;
	
	
	
};
4.Attributes【设置cpp文件】
// Fill out your copyright notice in the Description page of Project Settings.


#include "AttributeSet_basic.h"
#include "net/UnrealNetwork.h"

//设定函数里面的初始值，函数名字
UAttributeSet_basic::UAttributeSet_basic()
{
	Health = 100.f;
	MaxHealth = 100.f;
	Stamina = 100.f;
	MaxStamina = 100.f;
}

//设定调用时候有两个引脚
//获得永久性复制的函数
void UAttributeSet_basic::GetLifetimeReplicatedProps(TArray<FLifetimeProperty>& OutLifetimeProps) const
{
	//调用父项
	Super::GetLifetimeReplicatedProps(OutLifetimeProps);
	
	//设定重复复制的条件为COND_None，意味着其他角色可以重复复制
	DOREPLIFETIME_CONDITION_NOTIFY(UAttributeSet_basic, Health, COND_None, REPNOTIFY_Always);
	DOREPLIFETIME_CONDITION_NOTIFY(UAttributeSet_basic, MaxHealth, COND_None, REPNOTIFY_Always);
	DOREPLIFETIME_CONDITION_NOTIFY(UAttributeSet_basic, Stamina, COND_None, REPNOTIFY_Always);
	DOREPLIFETIME_CONDITION_NOTIFY(UAttributeSet_basic, MaxStamina, COND_None, REPNOTIFY_Always);
}
